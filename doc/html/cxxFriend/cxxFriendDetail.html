cxxFriendDetail

Container for child elements providing information about a friend function.

Description

The <cxxFriendDetail> element contains a mandatory <cxxFriendDefinition>child element. The optional child element <apiDesc> can be used to describe the operation of the friend function.

Example

<cxxFriendDetail>
    <cxxFriendDefinition>
        <cxxFriendDeclaredType>friend void</cxxFriendDeclaredType>
        <cxxFriendScopedName>Class_13969_00</cxxFriendScopedName>
        <cxxFriendPrototype>friend void friend_set(Class_13969_00 *, int)</cxxFriendPrototype>
        <cxxFriendNameLookup>friend_set(Class_13969_00 *,int)</cxxFriendNameLookup>
        <cxxFriendParameters>
            <cxxFriendParameter>
                <cxxFriendParameterDeclaredType>
                    <apiRelation href="class_class__13969__00.xml#class_class__13969__00">Class_13969_00</apiRelation> *</cxxFriendParameterDeclaredType>
                <cxxFriendParameterDefinitionName>p</cxxFriendParameterDefinitionName>
                <apiDefNote />
            </cxxFriendParameter>
            <cxxFriendParameter>
                <cxxFriendParameterDeclaredType>int</cxxFriendParameterDeclaredType>
                <cxxFriendParameterDefinitionName>i</cxxFriendParameterDefinitionName>
                <apiDefNote />
            </cxxFriendParameter>
        </cxxFriendParameters>
        <cxxFriendReturnDesc />
        <cxxFriendAPIItemLocation>
            <cxxFriendDeclarationFile name="filePath" value="T:/epoc32/include/defect/file_13969.h" />
            <cxxFriendDeclarationFileLine name="lineNumber" value="14" />

    </apiDesc>
</cxxFriendDetail>
